289  while (pThunkOrg->AddressOfData != 0)

293

512  public static bool IsManagedAssembly(string fileName)
513  {
514  using (Stream fileStream = new FileStream(fileName, FileMode.Open, FileAccess.Read))
515  using (BinaryReader binaryReader = new BinaryReader(fileStream))
516  {
517  if (fileStream.Length < 64)
518  {
519  return false;
520  }
521 
522  //PE Header starts @ 0x3C (60). Its a 4 byte header.
523  fileStream.Position = 0x3C;
524  uint peHeaderPointer = binaryReader.ReadUInt32();
525  if (peHeaderPointer == 0)
526  {
527  peHeaderPointer = 0x80;
528  }
529 
530  // Ensure there is at least enough room for the following structures:
531  // 24 byte PE Signature & Header
532  // 28 byte Standard Fields (24 bytes for PE32+)
533  // 68 byte NT Fields (88 bytes for PE32+)
534  // >= 128 byte Data Dictionary Table
535  if (peHeaderPointer > fileStream.Length - 256)
536  {
537  return false;
538  }
539 
540  // Check the PE signature. Should equal 'PE\0\0'.
541  fileStream.Position = peHeaderPointer;
542  uint peHeaderSignature = binaryReader.ReadUInt32();
543  if (peHeaderSignature != 0x00004550)
544  {
545  return false;
546  }
547 
548  // skip over the PEHeader fields
549  fileStream.Position += 20;
550 
551  const ushort PE32 = 0x10b;
552  const ushort PE32Plus = 0x20b;
553 
554  // Read PE magic number from Standard Fields to determine format.
555  var peFormat = binaryReader.ReadUInt16();
556  if (peFormat != PE32 && peFormat != PE32Plus)
557  {
558  return false;
559  }
560 
561  // Read the 15th Data Dictionary RVA field which contains the CLI header RVA.
562  // When this is non-zero then the file contains CLI data otherwise not.
563  ushort dataDictionaryStart = (ushort)(peHeaderPointer + (peFormat == PE32 ? 232 : 248));
564  fileStream.Position = dataDictionaryStart;
565 
566  uint cliHeaderRva = binaryReader.ReadUInt32();
567  if (cliHeaderRva == 0)
568  {
569  return false;
570  }
571 
572  return true;
573  }
574  }
